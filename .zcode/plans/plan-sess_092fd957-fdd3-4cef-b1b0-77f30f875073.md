## 计划:扩展 `MessageSending` 协议 — 内核发送层支持图片附件

### 目标
让用户消息支持图片附件,且附件状态(列表/增/删/清)由 `MessageSending` 协议统一管理,任何插件可通过 `kernel.messageSend` 共享同一份"待发附件"池。这为后续截图插件、剪贴板图片、文件拖拽等所有"图片入对话"场景提供统一底层。

### 2. `MessageSending` 协议扩展

```swift
@MainActor
public protocol MessageSending: ObservableObject {
    /// 现有
    var isSending: Bool { get }

    // MARK: - 新增:附件状态池

    /// 当前挂起、等待下次发送时随消息一起送出的附件
    var pendingAttachments: [LumiImageAttachment] { get }

    /// 添加一个附件到挂起池
    /// - 实现应在已包含相同 `id` 的情况下忽略(幂等)
    func addAttachment(_ attachment: LumiImageAttachment)

    /// 按 id 移除一个挂起附件;id 不存在则 no-op
    func removeAttachment(id: UUID)

    /// 清空所有挂起附件
    func clearAttachments()

    // MARK: - 发送

    /// 现有 text-only 便利方法(保留向后兼容)
    /// 默认实现:转发到带 attachments 的重载,attachments 为 `pendingAttachments` 的快照
    func sendMessage(_ content: String, conversationID: UUID?) async throws

    /// 新重载:文本 + 显式 attachments
    /// 实现应:trim content → 解析会话 → 把 attachments 序列化进 user message 的
    /// `metadata["imageAttachments"]`(JSON)→ 落库 → 触发 AgentTurnRunner
    /// - 调用此方法后是否清空 pendingAttachments 由实现决定(默认不清空,与旧行为一致)
    func sendMessage(
        _ content: String,
        imageAttachments: [LumiImageAttachment],
        conversationID: UUID?
    ) async throws

    func cancelCurrentRequest()
}

// 默认实现:text-only 转发 + 复用 pendingAttachments
public extension MessageSending {
    func sendMessage(_ content: String, conversationID: UUID?) async throws {
        try await sendMessage(
            content,
            imageAttachments: pendingAttachments,
            conversationID: conversationID
        )
    }
}
```

**契约要点**(写入 doc comment):
- `addAttachment` 是幂等的(同 id 不重复添加)
- `pendingAttachments` 是 `@Published`,UI 可直接 observe
- 新重载的 `imageAttachments` 是显式参数,**不**自动读取 `pendingAttachments`(允许一次性发送不同附件,例如 tool 结果图片)
- 现有 text-only 调用者(测试/mocks)无需修改,默认实现会转发
- `removeAttachment(id:)` id 不存在则 no-op

### 3. `MessageSender` 实现要点

```swift
public final class MessageSender: MessageSending, SuperLog {
    @Published public private(set) var isSending: Bool = false
    @Published public private(set) var pendingAttachments: [LumiImageAttachment] = []

    // 状态池 API
    public func addAttachment(_ attachment: LumiImageAttachment) {
        guard !pendingAttachments.contains(where: { $0.id == attachment.id }) else { return }
        pendingAttachments.append(attachment)
    }
    public func removeAttachment(id: UUID) {
        pendingAttachments.removeAll { $0.id == id }
    }
```

### 4. `AgentTurnRunner` 构造 LLM 请求时抽 metadata

`AgentTurnRunner.executeTurnLoop` 构造 `LumiLLMRequest` 那一行(大约 L130),改为:

```swift
// 抽出最后一条 user message 的 attachments(如有)
let pendingImages: [LumiImageAttachment]
if let lastUser = history.last(where: { $0.role == .user }),
   let json = lastUser.metadata["imageAttachments"],
   let data = json.data(using: .utf8),
   let decoded = try? JSONDecoder().decode([LumiImageAttachment].self, from: data) {
    pendingImages = decoded
} else {
    pendingImages = []
}

let request = LumiLLMRequest(
    messages: history,
    model: model,
    tools: tools,
    imageAttachments: pendingImages   // 新参数
)
```

这样下游 `LumiVisionMessageSupport.attachRequestImages` 已经能正确处理顶层 `imageAttachments`,无需再改 `MessageBridge`。
