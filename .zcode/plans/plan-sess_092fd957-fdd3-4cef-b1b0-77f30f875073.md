## 计划:ChatScreenshotPlugin — 区域截图 → 自动塞入附件预览条

### 目标
新建独立插件 `ChatScreenshotPlugin`,提供区域截图能力。用户在聊天输入区上方点击 📷 按钮(或按 ⌘⇧S),弹出全屏遮罩,鼠标拖选区域后截图,自动 `kernel.messageSend.addAttachment(...)` 注入附件挂起池,缩略图立即出现在 `ChatAttachmentPreviewPlugin` 的预览条上。

### 关键设计决策
- **位置**:`chatSectionActionBarItems`,`order = 81`,`.alwaysOn`
- **格式**:JPEG,长边 1920px,quality 0.85,`mimeType = "image/jpeg"`
- **快捷键**:走 `kernel.command?.registerCommand(menu:item:)`,不直接动 LumiFactory
- **入口**:`Notification.Name.lumiCaptureScreenshot`,由命令 post、按钮也 post
- **截图 API**:`SCScreenshotManager.captureImage(in:)`(macOS 14+),无 fallback
- **多屏**:`NSScreen.screens.reduce(.null) { $0.union($1.frame) }`,逐屏创建 NSPanel
- **Overlay**:`NSPanel(borderless)`,`level = .screenSaver`,`collectionBehavior = [.canJoinAllSpaces, .fullScreenAuxiliary]`
- **权限**:`CGPreflightScreenCaptureAccess()` false 时弹 NSAlert + 引导到 `x-apple.systempreferences:...Privacy_ScreenCapture`
- **下游**:`kernel.messageSend.addAttachment(LumiImageAttachment)`,不持有任何本地状态

### 1. `ChatScreenshotPlugin` 主类

```swift
@MainActor
public final class ChatScreenshotPlugin: LumiPlugin, SuperLog {
    public let id = "com.coffic.lumi.plugin.chat-screenshot"
    public let name = "Chat Screenshot"
    public let order = 81
    public let policy: LumiPluginPolicy = .alwaysOn

    private weak var kernel: LumiKernel?

    public init() {}

    public func onBoot(kernel: LumiKernel) async throws {
        self.kernel = kernel
    }

    public func onReady(kernel: LumiKernel) async throws {
        // 1. 监听截图触发通知
        NotificationCenter.default.addObserver(
            forName: .lumiCaptureScreenshot, object: nil, queue: .main
        ) { [weak self] _ in
            Task { @MainActor in
                await self?.triggerCapture(kernel: kernel)
            }
        }

        // 2. 注册 ⌘⇧S 全局命令(走 kernel.command 服务,无需改 LumiFactory)
        kernel.command?.registerCommand(
            menu: "Chat",
            item: CommandItem(
                id: "\(id).capture",
                title: String(localized: "Capture Screenshot", bundle: .module),
                shortcut: "s",
                modifiers: [.command, .shift]
            ) {
                NotificationCenter.default.post(name: .lumiCaptureScreenshot, object: nil)
            }
        )
    }

    public func chatSectionActionBarItems(kernel: LumiKernel) -> [ChatSectionActionBarItem] {
        [
            ChatSectionActionBarItem(id: "\(id).button") {
                ChatScreenshotButtonView(kernel: kernel)
            }
        ]
    }

    private func triggerCapture(kernel: LumiKernel) async {
        guard ScreenCapturePermissionPrompter.ensurePermission() else {
            ScreenCapturePermissionPrompter.presentAlert(openSettingsOnConfirm: true)
            return
        }
        guard let screenshot = try? await ScreenCaptureService.captureAllScreens() else {
            Self.logger.error("capture failed")
            return
        }
        ChatScreenshotState.shared.startSelection(
            image: screenshot.image,
            captureFrame: screenshot.frame,
            onComplete: { [weak kernel] pngData in
                guard let pngData else { return }
                let attachment = ScreenCaptureImageProcessor.makeAttachment(from: pngData)
                kernel?.messageSend?.addAttachment(attachment)
            }
        )
    }

    // LumiPlugin stubs
}
```

### 2. `ChatScreenshotState` (状态机)

```swift
@MainActor
public final class ChatScreenshotState {
    public static let shared = ChatScreenshotState()
    private var overlayController: ChatScreenshotOverlayController?

    private init() {}

    public func startSelection(
        image: CGImage,
        captureFrame: CGRect,
        onComplete: @escaping (Data?) -> Void
    ) {
        guard overlayController == nil else { return }
        let controller = ChatScreenshotOverlayController(
            image: image,
            captureFrame: captureFrame,
            onComplete: { [weak self] selection in
                self?.overlayController = nil
                guard let selection else { onComplete(nil); return }
                guard let cropped = Self.crop(image: image, captureFrame: captureFrame, selection: selection) else {
                    onComplete(nil); return
                }
                let pngData = NSBitmapImageRep(cgImage: cropped)
                    .representation(using: .jpeg, properties: [.compressionFactor: 0.85])!
                onComplete(pngData)
            }
        )
        overlayController = controller
        controller.show()
    }

    /// 纯几何函数(供单元测试)
    /// - 坐标系:captureFrame 与 selection 都在屏幕坐标(原点在左下)
    /// - Y 轴翻转:CGImage 原点在左上,所以 y = captureFrame.height - relativeY - selection.height
    public nonisolated static func crop(
        image: CGImage,
        captureFrame: CGRect,
        selection: CGRect
    ) -> CGImage? {
        let scaleX = CGFloat(image.width) / captureFrame.width
        let scaleY = CGFloat(image.height) / captureFrame.height
        let relativeX = selection.minX - captureFrame.minX
        let relativeY = selection.minY - captureFrame.minY
        let cropRect = CGRect(
            x: relativeX * scaleX,
            y: (captureFrame.height - relativeY - selection.height) * scaleY,
            width: selection.width * scaleX,
            height: selection.height * scaleY
        ).integral.intersection(CGRect(x: 0, y: 0, width: image.width, height: image.height))
        guard cropRect.width >= 1, cropRect.height >= 1 else { return nil }
        return image.cropping(to: cropRect)
    }
}
```

### 3. `ScreenCaptureService`

```swift
public enum ScreenCaptureService {
    public struct Result { let image: CGImage; let frame: CGRect }

    public static func captureAllScreens() async throws -> Result {
        let frame = NSScreen.screens.reduce(CGRect.null) { $0.union($1.frame) }
        guard !frame.isNull else { throw ScreenshotError.noScreens }
        let image: CGImage = try await withCheckedThrowingContinuation { cont in
            SCScreenshotManager.captureImage(in: frame) { image, error in
                if let error { cont.resume(throwing: error) }
                else if let image { cont.resume(returning: image) }
                else { cont.resume(throwing: ScreenshotError.captureFailed) }
            }
        }
        return Result(image: image, frame: frame)
    }

    public enum ScreenshotError: Error { case noScreens, captureFailed }
}
```

### 4. `ScreenCapturePermissionPrompter`

```swift
public enum ScreenCapturePermissionPrompter {
    public static func ensurePermission() -> Bool {
        CGPreflightScreenCaptureAccess()
    }

    public static func presentAlert(openSettingsOnConfirm: Bool) {
        let alert = NSAlert()
        alert.messageText = String(localized: "Screen Recording Permission Required", bundle: .module)
        alert.informativeText = String(localized: "Lumi needs Screen Recording permission to capture screenshots.", bundle: .module)
        alert.alertStyle = .warning
        alert.addButton(withTitle: String(localized: "Open System Settings", bundle: .module))
        alert.addButton(withTitle: String(localized: "Cancel", bundle: .module))
        if alert.runModal() == .alertFirstButtonReturn {
            NSWorkspace.shared.open(URL(string: "x-apple.systempreferences:com.apple.preference.security?Privacy_ScreenCapture")!)
        }
    }
}
```

### 5. `ScreenCaptureImageProcessor`(downscale + JPEG + base64)

```swift
public enum ScreenCaptureImageProcessor {
    public static let maxDimension: CGFloat = 1920
    public static let jpegQuality: CGFloat = 0.85

    public static func makeAttachment(from pngOrJpegData: Data) -> LumiImageAttachment {
        // 输入可能是任意编码(JPEG/PNG),统一转 CGImage → 长边 1920 → JPEG → base64
        guard let source = CGImageSourceCreateWithData(pngOrJpegData as CFData, nil),
              let original = CGImageSourceCreateImageAtIndex(source, 0, nil)
        else {
            // fallback:直接编码(保持原样)
            return LumiImageAttachment(
                mimeType: "image/jpeg",
                base64Data: pngOrJpegData.base64EncodedString()
            )
        }
        let scaled = downscale(original, maxDimension: maxDimension)
        guard let dest = CGImageDestinationCreateWithData(
            NSMutableData() as CFMutableData, "public.jpeg" as CFString, 1, nil
        ) else { return fallback(original: pngOrJpegData) }
        let mutableData = NSMutableData()
        // 实际需要先 downscale 再创建 destination,以下是骨架
        let attachment = LumiImageAttachment(
            mimeType: "image/jpeg",
            base64Data: mutableData as Data → base64,
            fileName: "screenshot-\(timestamp).jpg"
        )
        return attachment
    }

    private static func downscale(_ image: CGImage, maxDimension: CGFloat) -> CGImage {
        let w = CGFloat(image.width), h = CGFloat(image.height)
        let scale = min(1.0, maxDimension / max(w, h))
        guard scale < 1.0 else { return image }
        let newSize = CGSize(width: w * scale, height: h * scale)
        let context = CGContext(...)!
        context.draw(image, in: CGRect(origin: .zero, size: newSize))
        return context.makeImage()!
    }
}
```

### 不在本计划内
- **跨屏 selection 越界时的裁剪行为**(用户拖到第二屏边缘):已有 crop 函数 `.intersection(...)` 兜底
- **图片标注/马赛克**:截图后编辑功能超出范围
- **保存到磁盘**:截图只进 attachment pool,不落盘
- **`ChatScreenshotState` 改 ObservableObject**:进度条用其他方式(主类持有 `@Published var isPreparing` 即可,不污染 shared)
- **延迟 init 之后 TCC 被用户撤销**:下次点击时 CGPreflightScreenCaptureAccess 会重新返回 false,自动重弹引导
