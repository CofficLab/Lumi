### 详细设计

#### 1. `SettingsTab.swift` — 拆 id 与展示,新增 `SettingsTabID`

```swift
import Foundation
import LumiLocalizationKit

// 内置标签保持稳定 id(原 enum rawValue),便于未来 i18n 时不变 id
enum SettingsTab: String, CaseIterable, Identifiable {
    case general, appearance, plugins, about
    var id: String { rawValue }
    var title: String { LumiLocalization.string(...) }   // 不变
    var systemImage: String { ... }                       // 不变
}

/// 统一内置 + 插件 tab 的 selector,消除两份数据不同 id 类型的问题
enum SettingsTabID: Hashable {
    case builtin(SettingsTab)
    case plugin(String)            // = SettingsTabItem.id
}

struct SettingsTabDescriptor: Identifiable, Hashable {
    let id: SettingsTabID
    let title: String
    let systemImage: String
}
```

将原本 `ForEach(SettingsTab.allCases)` 与未来的 `ForEach(plugins)` 统一为 `ForEach(descriptors)`,避免双重 `Identifiable`。

#### 2. `SettingsView.swift` — 改观察 + 平铺 + 路由化详情

```swift
struct SettingsView: View {
    @LumiTheme private var theme
    @ObservedObject var kernel: LumiKernel                       // ← 由 let 改成 @ObservedObject
    @State private var selectedTab: SettingsTabID = .builtin(.general)

    /// 内置 + 插件 tab 合并平铺
    private var descriptors: [SettingsTabDescriptor] {
        var items: [SettingsTabDescriptor] = SettingsTab.allCases.map {
            SettingsTabDescriptor(
                id: .builtin($0),
                title: $0.title,
                systemImage: $0.systemImage
            )
        }
        for item in kernel.settings?.allSettingsTabItems ?? [] {
            items.append(SettingsTabDescriptor(
                id: .plugin(item.id),
                title: item.title,
                systemImage: item.systemImage
            ))
        }
        return items
    }

    private var sidebar: some View {
        AppSettingsSidebarContainer(width: 220) {
            VStack(alignment: .leading, spacing: 10) {
                SettingsSidebarHeaderView(kernel: kernel)
                AppSettingsDivider()
                VStack(spacing: 6) {
                    ForEach(descriptors) { desc in
                        AppSettingsSidebarItem(
                            title: desc.title,
                            systemImage: desc.systemImage,
                            isSelected: selectedTab == desc.id
                        ) { selectedTab = desc.id }
                    }
                }
                Spacer()
            }
        }
    }

    @ViewBuilder
    private var detail: some View {
        switch selectedTab {
        case .builtin(.general):    GeneralSettingsPage()
        case .builtin(.appearance): AppearanceSettingsPage(kernel: kernel)
        case .builtin(.plugins):    PluginSettingsPage(kernel: kernel)
        case .builtin(.about):      AboutPage()
        case .plugin(let id):
            if let item = kernel.settings?.allSettingsTabItems.first(where: { $0.id == id }) {
                item.makeContent()
                    .frame(maxWidth: .infinity, maxHeight: .infinity)
            } else {
                AppEmptyState(
                    icon: "questionmark.folder",
                    title: LumiLocalization.string("Settings tab unavailable", bundle: .module),
                    description: LumiLocalization.string("This plugin is no longer providing this tab.", bundle: .module)
                )
            }
        }
    }
}
```

设计要点:
- **观察 kernel 而不是 SettingsProviding**:与现有架构一致(`LumiPlugin`、logo、`viewContainer` 等都用 `@ObservedObject kernel`);settings 提供者会被 `registerService` 时自动转发到 kernel 的 `objectWillChange`(配合改动 #3)。
- **插件 tab 走 `item.makeContent()`** —— 该方法已经存在(`SettingsTabItem.makeContent() -> AnyView`),无需修改类型。
- **空状态**:启动时 settings tab 可能已被 unregister(罕见但可能),给用户一个友好提示而不是空白/崩溃。
- **`SettingsTabID` 持久化在 `@State` 里**:hashable,SwiftUI 直接比较。

#### 3. `DefaultSettingsProviding.swift` — 改 `@Published` + 显式发布

```swift
@MainActor
public final class DefaultSettingsProviding: SettingsProviding {
    @Published public private(set) var allSettingsTabItems: [SettingsTabItem] = []
    @Published public private(set) var allLLMProviderSettingsItems: [LLMProviderSettingsItem] = []

    private var settingsTabItems: [String: SettingsTabItem] = [:]
    private var settingsTabOrder: [String] = []
    private var llmProviderSettingsItems: [String: LLMProviderSettingsItem] = [:]
    private var llmProviderSettingsOrder: [String] = []

    public init() {}

    public func registerSettingsTabItem(_ item: SettingsTabItem) {
        if settingsTabItems[item.id] == nil { settingsTabOrder.append(item.id) }
        settingsTabItems[item.id] = item
        updateSortedSettingsTabs()                              // 已赋值 @Published 即触发 objectWillChange
    }
    public func unregisterSettingsTabItem(id: String) { ... }
    public func registerLLMProviderSettingsItem(_ item: LLMProviderSettingsItem) { ... }
    public func unregisterLLMProviderSettingsItem(providerID: String) { ... }

    private func updateSortedSettingsTabs() {
        allSettingsTabItems = settingsTabOrder.compactMap { settingsTabItems[$0] }
        // @Published wrapper 赋值时 SwiftUI 自动 send objectWillChange;
        // kernel 已经订阅了服务的 objectWillChange 并转发,所以 @ObservedObject kernel 会重渲。
    }
    private func updateSortedLLMProviderSettings() { ... }
}
```

`@Published` 不需要再手动 send。`LumiKernelContainer.subscribeToObjectWillChange` 已经把服务的 `objectWillChange` 转发到 `kernel.objectWillChange`,所以 `SettingsView` 的 `@ObservedObject var kernel` 自然会重渲并读到最新的 `allSettingsTabItems`。

> ⚠️ `SettingsProviding` 协议声明的 `var allSettingsTabItems` 没有 `{ get }` 之外的约束 — `@Published private(set) var` 在协议 witness 下没问题(protocol 只读,实现 private(set) 与之兼容)。
